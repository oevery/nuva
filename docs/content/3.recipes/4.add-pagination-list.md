## 目标

从 `shared`、`server/api`、`app/composables/apis` 到页面完成一个分页列表。

## 文件位置

```txt
app/api/types.ts
shared/api/user.ts
server/api/users.get.ts
app/composables/apis/user.ts
app/pages/users.vue
```

## 步骤

### 1. 定义分页协议

```ts
// app/api/types.ts
export type { PageParams, PageQuery, PageResult } from '#nuva/api'
```

### 2. 实现服务端接口

```ts
// server/api/users.get.ts
import type { User } from '#shared/api/user'
import type { PageResult } from '~/api/types'
import { ok } from '../utils/response'

export default defineEventHandler((event) => {
  const query = getQuery(event)
  const pageNum = Number(query.pageNum || 1)
  const pageSize = Number(query.pageSize || 10)

  return ok<PageResult<User>>({
    list: [],
    total: 0,
    pageNum,
    pageSize,
  })
})
```

### 3. 封装列表状态

```ts
// app/composables/apis/user.ts
import type { User, UserQuery } from '#shared/api/user'
import { reactive } from 'vue'

export function useUserApi() {
  return createResourceApi<User, Partial<User>, Partial<User>, UserQuery>({
    path: '/users',
  })
}

export function useUserTable() {
  const query = reactive<UserQuery>({
    keyword: undefined,
    status: undefined,
  })
  const userApi = useUserApi()

  return {
    query,
    ...usePageApi<User, UserQuery>(userApi.list, {
      query,
    }),
  }
}
```

### 4. 页面消费

```vue
<script setup lang="ts">
const {
  query,
  data: users,
  loading,
  page,
  pageSize,
  total,
  pageCount,
  refresh,
  reload,
} = useUserTable()
</script>
```

## 自定义分页字段

默认请求参数为 `pageNum` 和 `pageSize`，默认响应结构为 `PageResult<T>` 的 `list` 和 `total`。如果后端字段不同，在 `usePageApi()` 中覆盖：

```ts
usePageApi<User, UserQuery>(userApi.list, {
  query,
  pageField: 'page',
  pageSizeField: 'limit',
})
```

如果响应结构不同，让业务 API 返回对应响应类型，并在 `usePageApi()` 中传入 `data` 和 `total` 映射：

```ts
interface UserPageResponse {
  records: User[]
  totalCount: number
}

function useUserApi() {
  const http = useHttpClient()

  return {
    list: (params: PageParams & UserQuery) => http.Get<UserPageResponse>('/users', { params }),
  }
}

usePageApi<User, UserQuery, UserPageResponse>(userApi.list, {
  query,
  data: response => response.records,
  total: response => response.totalCount,
})
```

## 常见问题

- 页面层不要直接拼分页接口。
- SPA 外部接口的分页类型建议从 `#nuva/api` 转发到 `app/api/types.ts`；BFF/full-stack 共享协议才放 `shared/api`。
- `usePageApi()` 只是模板 helper，不是通用 dataProvider。
- `usePageApi()` 默认会过滤 `undefined`、`null` 和空字符串查询参数，可以设置 `cleanParams: false` 关闭。
- 如果后端分页字段不同，优先在 API composable 中适配，不要散落到页面。
